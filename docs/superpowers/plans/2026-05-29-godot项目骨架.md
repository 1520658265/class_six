# Godot 项目骨架与通用资产管线 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 把仓库根改造成 Godot 4.6 工程根，落地一份 manifest + 5 个烤点脚本 + 4 个 autoload helper + AssetIds 生成常量类，让业务代码用名字驱动加载美术资产。

**Tech Stack:** Godot 4.6（GDScript 2.0），GUT 9.x（unit testing），Python 3.13 + Pillow（一次性测试 fixture 生成），git。

**关联 spec:** `docs/superpowers/specs/2026-05-29-godot项目骨架-spec.md`

```gdscript
# tools/godot_bake/manifest_loader.gd
class_name ManifestLoader
extends RefCounted

const SUPPORTED_VERSIONS := [1]

static func load_file(path: String) -> Variant:
    if not FileAccess.file_exists(path):
        push_error("[manifest] file not found: %s" % path)
        return null
    var text := FileAccess.get_file_as_string(path)
    var json := JSON.new()
    var err := json.parse(text)
    if err != OK:
        push_error("[manifest] JSON parse error at line %d: %s" % [json.get_error_line(), json.get_error_message()])
        return null
    var data = json.data
    if typeof(data) != TYPE_DICTIONARY:
        push_error("[manifest] root must be object")
        return null
    return data

static func is_valid_version(data: Dictionary) -> bool:
    return data.get("version", 0) in SUPPORTED_VERSIONS

static func section(data: Dictionary, name: String) -> Dictionary:
    var s = data.get(name, {})
    if typeof(s) != TYPE_DICTIONARY:
        return {}
    return s
```

```gdscript
    var d = DirAccess.open("user://")

func _rmtree(path: String) -> void:
```

```gdscript
const PortraitSetCls = preload("res://scripts/shared/portrait_set.gd")

static func bake(portraits: Dictionary, output_dir: String) -> Dictionary:
    DirAccess.make_dir_recursive_absolute(output_dir)
    var ok := 0
    var skipped := 0
    var errors: Array[String] = []

    for entry_id in portraits.keys():
        var entry: Dictionary = portraits[entry_id]
        var sheet_path: String = entry.get("sheet", "")
        if not ResourceLoader.exists(sheet_path):
            errors.append("[ERR] portraits/%s: sheet not found %s" % [entry_id, sheet_path])
            skipped += 1
            continue
        var tex := load(sheet_path) as Texture2D

        var ps: Resource = PortraitSetCls.new()
        ps.character_id = entry_id

        var slice_mode: String = entry.get("slice_mode", "grid")
        var ok_entry := true

        if slice_mode == "regions":
            var regions: Dictionary = entry.get("regions", {})
            var sheet_rect := Rect2i(0, 0, tex.get_width(), tex.get_height())
            for expr_name in entry.get("expressions", {}).keys():
                var expr_entry: Dictionary = entry["expressions"][expr_name]
                var rect_data = regions.get(expr_entry.get("region", ""), null)
                if rect_data == null or rect_data.size() != 4:

                    ok_entry = false
                    break

                if r.size.x <= 0 or r.size.y <= 0 or not sheet_rect.encloses(r):
                    errors.append("[ERR] portraits/%s/%s: region out of bounds" % [entry_id, expr_name])
                    ok_entry = false
                    break
                var at := AtlasTexture.new()
                at.atlas = tex
                at.region = Rect2(r.position.x, r.position.y, r.size.x, r.size.y)
                ps.expressions[expr_name] = at

            var tile_h: int = entry.get("tile_h", 0)
            var rows: int = entry.get("rows", 0)
            var cols: int = entry.get("cols", 0)
            if tex.get_width() != tile_w * cols or tex.get_height() != tile_h * rows:
                errors.append("[ERR] portraits/%s: sheet size mismatch" % entry_id)
                skipped += 1
                continue
            for expr_name in entry.get("expressions", {}).keys():
                var ex: Dictionary = entry["expressions"][expr_name]
                var at := AtlasTexture.new()
                at.atlas = tex
                at.region = Rect2(ex.get("col", 0) * tile_w, ex.get("row", 0) * tile_h, tile_w, tile_h)

            continue
```

```gdscript
    DirAccess.make_dir_recursive_absolute(output_dir)
    var ok := 0
    var skipped := 0
```

```gdscript
# tools/godot_bake/bake_asset_ids.gd
class_name BakeAssetIds
extends RefCounted

## 把 manifest 的 4 个 section 转成 AssetIds 常量类。
## output_path 是单一 .gd 文件路径（不是目录）。
static func bake(manifest: Dictionary, output_path: String) -> Dictionary:
    var ok := 0
    var skipped := 0
    var errors: Array[String] = []

    var invalid_keys := _collect_invalid_keys(manifest)
    if invalid_keys.size() > 0:
        for key in invalid_keys:
            errors.append("[ERR] asset_ids: invalid key %s" % key)
        return { "ok": 0, "skipped": 1, "errors": errors }

    var lines: Array[String] = []
    lines.append("# AUTO-GENERATED by tools/godot_bake/bake_asset_ids.gd. DO NOT EDIT.")
    lines.append("# Last bake: %s" % Time.get_datetime_string_from_system())
    lines.append("")
    lines.append("class_name AssetIds")
    lines.append("extends RefCounted")
    lines.append("")

    var sprites: Dictionary = manifest.get("sprites", {})
    lines.append("class Sprites:")
    if sprites.is_empty():
        lines.append("    pass")
    else:
        for entry_id in sprites.keys():
            lines.append("    const %s := \"%s\"" % [_const_name(entry_id), entry_id])
    lines.append("")

    var portraits: Dictionary = manifest.get("portraits", {})
    lines.append("class Portraits:")
    if portraits.is_empty():
        lines.append("    pass")
    else:
        for entry_id in portraits.keys():
            lines.append("    class %s:" % _class_name(entry_id))
            var exprs: Dictionary = portraits[entry_id].get("expressions", {})
            if exprs.is_empty():
                lines.append("        pass")
            else:
                for expr_name in exprs.keys():
                    lines.append("        const %s := \"%s\"" % [_const_name(expr_name), expr_name])
    lines.append("")

    var vfx: Dictionary = manifest.get("vfx", {})
    lines.append("class Vfx:")
    if vfx.is_empty():
        lines.append("    pass")
    else:
        for entry_id in vfx.keys():
            lines.append("    const %s := \"%s\"" % [_const_name(entry_id), entry_id])
    lines.append("")

    var atlas_section: Dictionary = manifest.get("atlas", {})
    lines.append("class Atlas:")
    if atlas_section.is_empty():
        lines.append("    pass")
    else:
        for entry_id in atlas_section.keys():
            lines.append("    class %s:" % _class_name(entry_id))
            var items: Dictionary = atlas_section[entry_id].get("items", {})
            if items.is_empty():
                lines.append("        pass")
            else:
                for item_name in items.keys():
                    lines.append("        const %s := \"%s\"" % [_const_name(item_name), item_name])
    lines.append("")

    var dir := output_path.get_base_dir()
    DirAccess.make_dir_recursive_absolute(dir)
    var f := FileAccess.open(output_path, FileAccess.WRITE)
    if f == null:
        errors.append("[ERR] asset_ids: cannot open %s" % output_path)
        return { "ok": 0, "skipped": 1, "errors": errors }
    f.store_string("\n".join(lines))
    f.close()

    print("[OK] asset_ids -> %s" % output_path)
    return { "ok": 1, "skipped": 0, "errors": errors }

static func _collect_invalid_keys(manifest: Dictionary) -> Array[String]:
    var invalid: Array[String] = []
    for section_name in ["sprites", "portraits", "vfx", "atlas"]:
        var section: Dictionary = manifest.get(section_name, {})
        for entry_id in section.keys():
            _append_invalid(invalid, "%s.%s" % [section_name, entry_id], entry_id)
            if section_name == "portraits":
                var exprs: Dictionary = section[entry_id].get("expressions", {})
                for expr_name in exprs.keys():
                    _append_invalid(invalid, "%s.%s.%s" % [section_name, entry_id, expr_name], expr_name)
            elif section_name == "atlas":
                var items: Dictionary = section[entry_id].get("items", {})
                for item_name in items.keys():
                    _append_invalid(invalid, "%s.%s.%s" % [section_name, entry_id, item_name], item_name)
    return invalid

static func _append_invalid(invalid: Array[String], path: String, key: String) -> void:
    if not _is_snake_id(key):
        invalid.append(path)

static func _is_snake_id(key: String) -> bool:
    if key.is_empty():
        return false
    var re := RegEx.new()
    re.compile("^[a-z][a-z0-9_]*$")
    return re.search(key) != null

static func _const_name(snake: String) -> String:
    return snake.to_upper()

static func _class_name(snake: String) -> String:
    var out := ""
    for part in snake.split("_"):
        if part.length() > 0:
            out += part.substr(0, 1).to_upper() + part.substr(1)
    return out
```

```gdscript
var resource_dir: String = "res://resources/vfx"

func spawn(vfx_name: String, pos: Vector2, direction: float = 1.0) -> AnimatedSprite2D:
    var clip := _load_clip(vfx_name)
    if clip == null:
        return null
    var sprite := AnimatedSprite2D.new()
    sprite.sprite_frames = clip.frames
    sprite.global_position = pos + Vector2(clip.offset.x * direction, clip.offset.y)
    sprite.scale = clip.scale_from
    if direction < 0:
        sprite.flip_h = true

    var parent := get_tree().current_scene
    if parent == null:
        parent = self  # in unit tests, attach to helper itself
    parent.add_child(sprite)
    sprite.play("default")

    var tween := sprite.create_tween()
    tween.tween_property(sprite, "scale", clip.scale_to, clip.scale_dur).set_ease(Tween.EASE_OUT)
    tween.parallel().tween_property(sprite, "modulate:a", clip.alpha_to, clip.alpha_dur).from(clip.alpha_from)
    tween.tween_callback(sprite.queue_free)
    return sprite

func _load_clip(vfx_name: String) -> VfxClip:
```

```gdscript
        push_error("Missing atlas item: %s/%s" % [category, item_name])
        return null
    return atlas.items[item_name]

func get_set(category: String) -> AtlasSet:
    return _get_set(category)

func _get_set(category: String) -> AtlasSet:
    if _cache.has(category):
        return _cache[category]
    var path := "%s/%s.tres" % [resource_dir, category]
    if not ResourceLoader.exists(path):
        push_error("Missing AtlasSet: %s" % path)
        return null
    var res := load(path) as AtlasSet
    if res == null:
        push_error("Invalid AtlasSet at %s" % path)
        return null
    _cache[category] = res
    return res
```

# tools/godot_bake/

把 `assets/art/` 的原始 sheet 烤成 Godot 用的 `.tres` 资源 + 生成 `AssetIds` 常量类。

参考完整 spec：`docs/superpowers/specs/2026-05-29-godot项目骨架-spec.md`

## 加新资产的步骤

1. 把 sheet 放到 `assets/art/...` 下，用规范化文件名（不带 `_v数字`）
2. 在 `tools/godot_bake/manifest.json` 加一段 entry
3. 跑烤点：
   ```bash
   godot --headless --path . -s tools/godot_bake/bake_all.gd
   ```
4. 把 `tools/godot_bake/manifest.json`、`resources/**/*.tres`、`scripts/autoload/asset_ids.gd` 一并 commit

第 3 步以后业务代码可以直接用 `Sprites.walk_frames(AssetIds.Sprites.YOUR_NEW_ID)` 引用。

## 退出码

- 0：全部成功
- 1：部分跳过（某些 entry 有问题，但其他已成功落盘）
- 2：致命（manifest.json 解析失败 / 版本不支持）

## manifest.json 字段说明

详见 spec 第五节。简要：

| section | 用途 | 输出资源 |
|---|---|---|
| `sprites` | 角色行走/战斗动画 | SpriteFrames |
| `portraits` | 对话肖像表情贴图 | PortraitSet |
| `vfx` | 战斗特效（含 tween preset） | VfxClip |
| `atlas` | 道具/UI/场景图标贴图 | AtlasSet |

每条 entry 都需要 `sheet` 路径 + 切片信息（`grid` 或 `regions` 模式）。

## 常见错误

| 现象 | 原因 | 处理 |
|---|---|---|
| `sheet not found` | manifest 路径写错或资产没复制 | 检查 `sheet` 字段是否 `res://` 起始且文件存在 |
| `sheet size mismatch` | `tile_w * cols ≠ sheet.width` | 用 PIL 检查实际尺寸，回头改 `tile_w/tile_h` |
| `region out of bounds` | regions 模式矩形超出 sheet | 检查 `[x,y,w,h]` 是否在 sheet 内 |
| 业务代码报 `Missing SpriteFrames` | 没跑烤点或 entry 没 commit | 跑 `bake_all.gd` 并 commit 产物 |
